批处理并发

简单并发推荐

function batchCurlRequests(array $urls, int $batchSize = 8): array
{
    dump(microtime(true));

    $results = [];

    // 将原始 URL 保留索引进行分组
    $chunks = array_chunk($urls, $batchSize, true); // 保留原始键名

    foreach ($chunks as $chunk) {
        $mh = curl_multi_init();
        $chs = [];
        $map = []; // curl句柄ID => 原始索引

        // 初始化本批次 curl 连接
        foreach ($chunk as $index => $url) {
            $ch = curl_init($url);
            curl_setopt_array($ch, [
                CURLOPT_RETURNTRANSFER => true,
                CURLOPT_TIMEOUT => 20,
                CURLOPT_CONNECTTIMEOUT => 5,
                CURLOPT_FOLLOWLOCATION => true,
            ]);
            curl_multi_add_handle($mh, $ch);
            $chs[(int)$ch] = $ch;
            $map[(int)$ch] = $index;
        }

        $active = null;

        // 执行当前批次
        do {
            do {
                // 监控句柄状态，等待就绪
                $status = curl_multi_exec($mh, $active);
            } while ($status === CURLM_CALL_MULTI_PERFORM);

            // 如果仍有未完成的请求，调用curl_multi_select等待，避免空转
            if ($active > 0) {
                curl_multi_select($mh, 1.0);
            }

            while ($info = curl_multi_info_read($mh)) {
                $ch = $info['handle'];
                $index = $map[(int)$ch];

                $results[$index] = [
                    'http_code' => curl_getinfo($ch, CURLINFO_HTTP_CODE),
                    'body' => curl_multi_getcontent($ch),
                    'url' => curl_getinfo($ch, CURLINFO_EFFECTIVE_URL),
                    'error' => curl_errno($ch) ? curl_error($ch) : null,
                ];

                curl_multi_remove_handle($mh, $ch);
                curl_close($ch);
                unset($chs[(int)$ch], $map[(int)$ch]);
            }
        } while ($active && $status === CURLM_OK);

        curl_multi_close($mh);
    }

    dump(microtime(true));

    // 保证结果顺序与原始 URL 顺序一致
    ksort($results);
    return array_values($results);
}

滑动窗口式并发

追求极致并发推荐


function batchCurlRequests2(array $urls, int $batchSize = 8): array
{
    dump(microtime(true));

    $mh = curl_multi_init();
    $results = [];
    $chs = [];
    $map = []; // curl句柄ID => 索引
    $pending = [];

    foreach ($urls as $i => $url) {
        $pending[] = ['index' => $i, 'url' => $url];
    }

    $active = null;

    // 初始化前 batchSize 个请求（滑动窗口）| 预填充初始窗口
    while (count($chs) < $batchSize && !empty($pending)) {
        $task = array_shift($pending);
        $ch = curl_init($task['url']);
        curl_setopt_array($ch, [
            CURLOPT_RETURNTRANSFER => true,
            CURLOPT_TIMEOUT => 20,
            CURLOPT_CONNECTTIMEOUT => 5,
            CURLOPT_FOLLOWLOCATION => true,
        ]);
        curl_multi_add_handle($mh, $ch);
        $chs[(int)$ch] = $ch;
        $map[(int)$ch] = $task['index'];
    }

    do {
        // 执行 curl
        do {
            $status = curl_multi_exec($mh, $active);
        } while ($status === CURLM_CALL_MULTI_PERFORM);

        // 等待活动连接
        if ($active > 0) {
            curl_multi_select($mh, 1.0);
        }

        // 用 curl_multi_info_read 检查哪个句柄完成，处理结果，关闭它
        while ($info = curl_multi_info_read($mh)) {
            $ch = $info['handle'];
            $index = $map[(int)$ch];

            $results[$index] = [
                'http_code' => curl_getinfo($ch, CURLINFO_HTTP_CODE),
                'body' => curl_multi_getcontent($ch),
                'url' => curl_getinfo($ch, CURLINFO_EFFECTIVE_URL),
                'error' => curl_errno($ch) ? curl_error($ch) : null,
            ];

            curl_multi_remove_handle($mh, $ch);
            curl_close($ch);
            unset($chs[(int)$ch], $map[(int)$ch]);

            // 填充下一个请求，保持并发窗口
            if (!empty($pending)) {
                $task = array_shift($pending);
                $newCh = curl_init($task['url']);
                curl_setopt_array($newCh, [
                    CURLOPT_RETURNTRANSFER => true,
                    CURLOPT_TIMEOUT => 20,
                    CURLOPT_CONNECTTIMEOUT => 5,
                    CURLOPT_FOLLOWLOCATION => true,
                ]);
                curl_multi_add_handle($mh, $newCh);
                $chs[(int)$newCh] = $newCh;
                $map[(int)$newCh] = $task['index'];
            }
        }
    } while ($active || !empty($chs));

    curl_multi_close($mh);
    dump(microtime(true));

    // 按索引顺序排序，确保结果与输入一致
    ksort($results);
    return array_values($results);
}

两种并发方式对比

对比项 批处理并发 滑动窗口式并发
并发控制方式 每批次 batchSize 个并发请求，等待全部完成再进入下一批 每次最多并发 batchSize 个请求，完成一个立刻加一个，始终保持最大并发
curl_multi_init 创建次数 每一批创建一次 整体只创建一次
实现复杂度 简单易懂，逻辑清晰 稍复杂，需更细致管理并发执行状态
curl_multi_handle 生命周期 每批新建/销毁一次 全局使用一个 multi handle
内存占用 一次只处理少量句柄，更省内存 内存占用低：处理完立即清理句柄
易错风险 低，流程清晰 中，容易出现 select、exec、active 不协调的问题
资源利用率 有空转（等待整个 batch 完成才执行下一个） 更高效，完成一个马上补下一个
总请求耗时 相对较慢（受最慢的每一批影响） 更快（并发池实时填充）
